# OpenDomain AI 一键部署配置示例（MVP版，后续完善）
# 详细部署步骤参考 docs/roadmap.md
version: '3.8'

services:
  backend:
    build: ./backend
    restart: always
    env_file: ./backend/.env
    depends_on:
      - postgres
      - redis

  postgres:
    image: postgres:15-alpine
    restart: always
    environment:
      - POSTGRES_USER=opendomain
      - POSTGRES_PASSWORD=opendomain123
      - POSTGRES_DB=opendomain
    volumes:
      - postgres_data:/var/lib/postgresql/data

  redis:
    image: redis:7-alpine
    restart: always
    volumes:
      - redis_data:/data

volumes:
  postgres_data:
  redis_data:
